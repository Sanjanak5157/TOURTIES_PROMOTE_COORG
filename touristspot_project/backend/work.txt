import React, {useState, useEffect} from 'react';
import './styles.css';
import {fetchPlaces} from './api';
import PlaceGrid from './components/PlaceGrid';
import PlaceDetail from './components/PlaceDetail';

export default function App(){
  const [places, setPlaces] = useState([]);
  const [selected, setSelected] = useState(null);
  const [q, setQ] = useState('');

  useEffect(()=>{ fetchPlaces().then(setPlaces).catch(()=>setPlaces([])) }, []);

  const filtered = places.filter(p => (p.name || '').toLowerCase().includes(q.toLowerCase()));

  return (
    <div className="container">
      <h1>Tourist Spots</h1>
      {!selected && (
        <div>
          <div className="search-row">
            <input value={q} onChange={e=>setQ(e.target.value)} placeholder="Search" />
            <button className="reviews">Reviews</button>
          </div>
          <h3>List of Places</h3>
          <PlaceGrid places={filtered} onOpen={(p)=>setSelected(p)} />
        </div>
      )}

      {selected && (
        <PlaceDetail place={selected} onBack={()=>setSelected(null)} />
      )}
    </div>
  )
}

# ✅ **Option 1: Load places from backend API**

If your backend returns:

```
[
  { id: 1, name: "Beach Paradise", image: "/images/beach.jpg" },
  ...
]
```

Then your **App.js should be like this:**

```jsx
import React, { useState, useEffect } from "react";
import PlaceGrid from "./components/PlaceGrid";

export default function App() {
  const [places, setPlaces] = useState([]);
  const [selected, setSelected] = useState(null);

  useEffect(() => {
    fetch("http://localhost:5000/api/places")  // your backend URL
      .then(res => res.json())
      .then(data => setPlaces(data))
      .catch(() => setPlaces([]));
  }, []);

  return (
    <div>
      {!selected && (
        <PlaceGrid places={places} onOpen={(p) => setSelected(p)} />
      )}

      {selected && (
        <div style={{ padding: "20px" }}>
          <h2>{selected.name}</h2>
          <img
            src={selected.image}
            alt={selected.name}
            style={{ width: "300px", borderRadius: "10px" }}
          />
          <br/><br/>
          <button onClick={() => setSelected(null)}>Back</button>
        </div>
      )}
    </div>
  );
}
```

# ✅ **Option 2: Load places from a separate file (`places.js`)**

Create:

```
src/data/places.js
```

Put inside:

```js
const places = [
  { id: 1, name: "Beach Paradise", image: "/images/beach.jpg" },
  { id: 2, name: "Mountain View", image: "/images/mountain.jpg" },
  { id: 3, name: "Forest Escape", image: "/images/forest.jpg" }
];

export default places;
```

Then **App.js becomes clean**:

```jsx
import React, { useState } from "react";
import PlaceGrid from "./components/PlaceGrid";
import places from "./data/places";

export default function App() {
  const [selected, setSelected] = useState(null);

  return (
    <div>
      {!selected && (
        <PlaceGrid places={places} onOpen={(p) => setSelected(p)} />

        <div style={{ padding: "20px" }}>
          <h2>{selected.name}</h2>
          <img
            src={selected.image}
            alt={selected.name}
            style={{ width: "300px", borderRadius: "10px" }}
          />
          <br/><br/>
          <button onClick={() => setSelected(null)}>Back</button>
        </div>
      )}
    </div>
  );
}
```
